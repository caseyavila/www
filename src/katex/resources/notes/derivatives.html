Derivatives

\(\dfrac{d}{dx}\left[\left|{u}\right|\right] = \dfrac{u}{\left|u\right|}(u')\)

\(\dfrac{d}{dx}\left[\ln{u}\right] = \dfrac{u'}{u}\)

\(\dfrac{d}{dx}\left[e^u\right] = e^u u'\)

\(\dfrac{d}{dx}\left[\log_{a} u\right] = \dfrac{u'}{u\ln{a}}\)

\(\dfrac{d}{dx}\left[a^u\right] = \left(\ln{a}\right)a^u u'\)

\(\dfrac{d}{dx}\left[\sin{u}\right] = \left(\cos{u}\right)u'\)

\(\dfrac{d}{dx}\left[\cos{u}\right] = -\left(\sin{u}\right)u'\)

\(\dfrac{d}{dx}\left[\tan{u}\right] = \left(\sec^2{u}\right)u'\)

\(\dfrac{d}{dx}\left[\cot{u}\right] = -\left(\csc^2{u}\right)u'\)

\(\dfrac{d}{dx}\left[\sec{u}\right] = \left(\sec{u}\tan{u}\right)u'\)

\(\dfrac{d}{dx}\left[\csc{u}\right] = -\left(\csc{u}\cot{u}\right)u'\)

\(\dfrac{d}{dx}\left[\arcsin{u}\right] = \dfrac{u'}{\sqrt{1-u^2}}\)

\(\dfrac{d}{dx}\left[\arccos{u}\right] = \dfrac{-u'}{\sqrt{1-u^2}}\)

\(\dfrac{d}{dx}\left[\arctan{u}\right] = \dfrac{u'}{1+u^2}\)

\(\dfrac{d}{dx}\left[\operatorname{arccot}u\right] = \dfrac{-u'}{1+u^2}\)

\(\dfrac{d}{dx}\left[\operatorname{arcsec}u\right] = \dfrac{u'}{\left|u\right|\sqrt{u^2 +1}}\)

\(\dfrac{d}{dx}\left[\operatorname{arccsc}u\right] = \dfrac{-u'}{\left|u\right|\sqrt{u^2 +1}}\)
